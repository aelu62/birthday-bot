import discord
from discord.ext import commands
import os
from pymongo import MongoClient

# --- ⚙️ 디스코드 봇 기본 설정 ⚙️ ---
intents = discord.Intents.default()
intents.message_content = True
bot = commands.Bot(command_prefix='!', intents=intents)

# --- ⚙️ 몽고DB(비밀 금고) 설정 ⚙️ ---
# 클라우드타입 환경변수에 적어둔 MONGO_URL을 가져옵니다.
MONGO_URL = os.environ.get('MONGO_URL')

# 몽고DB에 연결! (데이터베이스 이름: birthday_db, 폴더 이름: users)
client = MongoClient(MONGO_URL)
db = client['birthday_db']
collection = db['users']

# --- 🚀 봇이 켜졌을 때 실행되는 부분 ---
@bot.event
async def on_ready():
    print(f'✅ 로그인 성공: {bot.user.name}')
    print('✅ 몽고DB 연결 및 봇 실행 완료! 이제 데이터가 영구 저장됩니다!')

# --- 🎂 생일 등록 명령어 ---
@bot.command(name='생일등록')
async def register_birthday(ctx, date: str):
    user_id = str(ctx.author.id)
    
    # 몽고DB에 저장하거나 업데이트 (upsert=True: 없으면 새로 만들고, 있으면 수정)
    collection.update_one(
        {'_id': user_id}, 
        {'$set': {'date': date, 'name': ctx.author.name}}, 
        upsert=True
    )
    await ctx.send(f'🎉 {ctx.author.name}님의 생일이 **{date}**로 안전한 금고(MongoDB)에 저장되었습니다!')

# --- 🎂 생일 확인 명령어 ---
@bot.command(name='생일확인')
async def check_birthday(ctx):
    user_id = str(ctx.author.id)
    
    # 몽고DB에서 내 데이터 찾기
    user_data = collection.find_one({'_id': user_id})
    
    if user_data:
        await ctx.send(f'📅 {ctx.author.name}님의 등록된 생일은 **{user_data["date"]}** 입니다!')
    else:
        await ctx.send(f'😥 아직 등록된 생일이 없어요. `!생일등록 MM-DD` 명령어로 등록해 주세요!')

# --- 🔑 가장 중요한 마지막 줄! (디스코드 봇 켜기) ---
# 클라우드타입에 적어둔 BOT_TOKEN을 가져와서 실행합니다.
TOKEN = os.environ.get('BOT_TOKEN')
bot.run(TOKEN)
