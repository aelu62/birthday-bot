import discord
from discord.ext import commands, tasks
import os
from pymongo import MongoClient
from datetime import datetime, time, timezone, timedelta

# --- ⚙️ 기본 및 시간 설정 ---
KST = timezone(timedelta(hours=9))
intents = discord.Intents.default()
intents.message_content = True
bot = commands.Bot(command_prefix='!', intents=intents)

# --- ⚙️ 몽고DB(비밀 금고) 연결 ---
MONGO_URL = os.environ.get('MONGO_URL')
client = MongoClient(MONGO_URL)
db = client['birthday_db']
collection = db['users']
settings_collection = db['settings']

# --- 🚀 봇이 켜졌을 때 ---
@bot.event
async def on_ready():
    print(f'✅ 로그인 성공: {bot.user.name}')
    if not check_birthday_loop.is_running():
        check_birthday_loop.start()

# --- 🛠️ 생일 축하 채널 설정 (👑 관리자 전용) ---
@bot.command(name='생일채널설정')
@commands.has_permissions(administrator=True)
async def set_birthday_channel(ctx, channel: discord.TextChannel = None):
    if channel is None:
        channel = ctx.channel
    settings_collection.update_one(
        {'_id': str(ctx.guild.id)},
        {'$set': {'channel_id': str(channel.id)}},
        upsert=True
    )
    await ctx.send(f'📢 띠링! 이 서버의 자동 생일 축하 메시지는 {channel.mention} 채널에만 올라옵니다! 🛠️')

@set_birthday_channel.error
async def set_birthday_channel_error(ctx, error):
    if isinstance(error, commands.MissingPermissions):
        await ctx.send("🚫 삐빅! 이 명령어는 서버 관리자만 사용할 수 있어요!")

# --- 🎂 생일 등록 (서버별 독립 저장!) ---
@bot.command(name='생일등록')
async def register_birthday(ctx, date: str):
    guild_id = str(ctx.guild.id)
    user_id = str(ctx.author.id)
    doc_id = f"{guild_id}_{user_id}" # 서버아이디와 유저아이디를 합쳐서 고유 열쇠 만들기!
    
    existing_user = collection.find_one({'_id': doc_id})
    if existing_user:
        await ctx.send(f'⚠️ 이미 이 서버에 생일(**{existing_user["date"]}**)이 등록되어 있어요! 변경을 원하시면 `!생일변경 MM-DD`를 사용해 주세요.')
        return
        
    collection.update_one(
        {'_id': doc_id}, 
        {'$set': {'guild_id': guild_id, 'user_id': user_id, 'date': date, 'name': ctx.author.name}}, 
        upsert=True
    )
    await ctx.send(f'🎉 {ctx.author.name}님의 생일(**{date}**)이 **{ctx.guild.name}** 에 저장되었습니다! ')

# --- 🔄 생일 변경 ---
@bot.command(name='생일변경')
async def change_birthday(ctx, date: str):
    guild_id = str(ctx.guild.id)
    user_id = str(ctx.author.id)
    doc_id = f"{guild_id}_{user_id}"
    
    existing_user = collection.find_one({'_id': doc_id})
    if not existing_user:
        await ctx.send(f'😥 이 서버에는 아직 등록된 생일이 없어요. `!생일등록 MM-DD` 명령어로 먼저 등록해 주세요!')
        return
        
    collection.update_one(
        {'_id': doc_id}, 
        {'$set': {'date': date, 'name': ctx.author.name}}
    )
    await ctx.send(f'🔄 {ctx.author.name}님의 생일이 **{date}**(으)로 성공적으로 변경되었습니다!')

# --- 🎂 생일 확인 (디데이 타이머 포함) ---
@bot.command(name='생일확인')
async def check_birthday(ctx):
    guild_id = str(ctx.guild.id)
    user_id = str(ctx.author.id)
    doc_id = f"{guild_id}_{user_id}"
    
    user_data = collection.find_one({'_id': doc_id})
    
    if user_data:
        bday_str = user_data["date"]
        await ctx.send(f'📅 {ctx.author.name}님의 등록된 생일은 **{bday_str}** 입니다!')
        
        now = datetime.now(KST)
        today_date = now.date()
        
        try:
            b_month, b_day = map(int, bday_str.split('-'))
            next_bday = datetime(now.year, b_month, b_day, tzinfo=KST).date()
        except ValueError:
            next_bday = datetime(now.year, 3, 1, tzinfo=KST).date()

        if next_bday < today_date:
            try:
                next_bday = datetime(now.year + 1, b_month, b_day, tzinfo=KST).date()
            except ValueError:
                next_bday = datetime(now.year + 1, 3, 1, tzinfo=KST).date()
        
        days_left = (next_bday - today_date).days
        
        if days_left == 0:
            await ctx.send(f'🎉 앗! 오늘이 바로 {ctx.author.name}님의 생일이네요! 정말정말 축하해요!! 🎂🎁🎊')
        elif days_left == 1:
            midnight_tomorrow = datetime.combine(next_bday, time(0, 0), tzinfo=KST)
            time_diff = midnight_tomorrow - now
            hours_left = int(time_diff.total_seconds() // 3600)
            await ctx.send(f'두근두근... 😳 내일이 드디어 생일이에요! 자정까지 딱 **{hours_left}시간** 남았습니다! ⏳')
        else:
            await ctx.send(f'생일까지 **D-{days_left}일** 남았어요! 🗓️⭐')
    else:
        await ctx.send(f'😥 이 서버에는 아직 등록된 생일이 없어요. `!생일등록 MM-DD` 명령어로 등록해 주세요!')

# --- 📜 생일 목록 (현재 서버의 목록만 보여주기!) ---
@bot.command(name='생일목록')
async def list_birthdays(ctx):
    guild_id = str(ctx.guild.id)
    # 몽고DB에서 '이 서버(guild_id)'에 해당하는 사람만 찾아서 날짜순 정렬!
    users = collection.find({'guild_id': guild_id}).sort("date", 1)
    
    if collection.count_documents({'guild_id': guild_id}) == 0:
        await ctx.send("🎂 이 서버에는 아직 등록된 생일이 없어요!")
        return
    
    msg = f"🎂 **{ctx.guild.name} 서버 생일 목록** 🎂\n"
    msg += "━━━━━━━━━━━━━━━━━━\n"
    for user in users:
        msg += f"• **{user['date']}** : {user['name']}님\n"
    await ctx.send(msg)

# --- ⏰ 매일 자정에 자동 생일 축하 (각 서버별로 따로따로!) ---
@tasks.loop(time=time(hour=0, minute=0, tzinfo=KST))
async def check_birthday_loop():
    today_str = datetime.now(KST).strftime("%m-%d")
    
    for guild in bot.guilds:
        # 이 서버에서 오늘 생일인 사람만 쏙쏙 뽑아오기
        birthday_users = list(collection.find({'guild_id': str(guild.id), 'date': today_str}))
        
        if len(birthday_users) > 0:
            guild_setting = settings_collection.find_one({'_id': str(guild.id)})
            channel = None
            if guild_setting and 'channel_id' in guild_setting:
                channel = guild.get_channel(int(guild_setting['channel_id']))
            else:
                channel = guild.system_channel
                if not channel:
                    channel = guild.text_channels[0]
            
            if channel:
                for user in birthday_users:
                    # user_id로 멘션하기
                    await channel.send(f"🎉 **축하합니다!** 오늘은 <@{user['user_id']}>님의 생일이에요! 모두 모여서 축하해 주세요! 🎂🎁")

# --- 🔑 디스코드 봇 켜기 ---
TOKEN = os.environ.get('BOT_TOKEN')
bot.run(TOKEN)
