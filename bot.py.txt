import discord
from discord.ext import commands, tasks
import os
from pymongo import MongoClient
from datetime import datetime, time, timezone, timedelta

# --- ⚙️ 기본 및 시간 설정 ---
KST = timezone(timedelta(hours=9)) # 한국 시간 기준
intents = discord.Intents.default()
intents.message_content = True
bot = commands.Bot(command_prefix='!', intents=intents)

# --- ⚙️ 몽고DB(비밀 금고) 연결 ---
MONGO_URL = os.environ.get('MONGO_URL')
client = MongoClient(MONGO_URL)
db = client['birthday_db']
collection = db['users']

# --- 🚀 봇이 켜졌을 때 ---
@bot.event
async def on_ready():
    print(f'✅ 로그인 성공: {bot.user.name}')
    if not check_birthday_loop.is_running():
        check_birthday_loop.start() # 봇이 켜지면 매일 생일 확인하는 타이머도 같이 시작!

# --- 🎂 생일 등록 (몽고DB 실시간 확인 기능 추가!) ---
@bot.command(name='생일등록')
async def register_birthday(ctx, date: str):
    user_id = str(ctx.author.id)
    
    # 1. 먼저 몽고DB 서버에 데이터를 쓱 밀어 넣습니다.
    collection.update_one(
        {'_id': user_id}, 
        {'$set': {'date': date, 'name': ctx.author.name}}, 
        upsert=True
    )
    
    # 2. 몽고DB 서버에서 방금 넣은 내 데이터가 진짜 있는지 다시 꺼내서 검사합니다! 👀
    check_data = collection.find_one({'_id': user_id})
    
    # 3. 확인 결과에 따라 디스코드에 메시지 보내기
    if check_data:
        await ctx.send(f'🎉 {check_data["name"]}님의 생일(**{check_data["date"]}**)이 저장되었습니다! ')
    else:
        await ctx.send('⚠️ 에러: 몽고DB 서버에 저장을 시도했지만, 서버에서 데이터를 찾지 못했습니다. 다시 시도해 주세요!')

# --- 🎂 생일 확인 (내 생일) ---
@bot.command(name='생일확인')
async def check_birthday(ctx):
    user_id = str(ctx.author.id)
    
    # 몽고DB에서 내 데이터 찾기
    user_data = collection.find_one({'_id': user_id})
    
    if user_data:
        # 1. 기본적으로 며칠인지 알려주기
        await ctx.send(f'📅 {ctx.author.name}님의 등록된 생일은 **{user_data["date"]}** 입니다!')
        
        # 2. 오늘 날짜 구하기 (한국 시간 기준)
        today_str = datetime.now(KST).strftime("%m-%d")
        
        # 3. 만약 오늘이 생일이라면? 깜짝 축하 파티! 🎉
        if user_data["date"] == today_str:
            await ctx.send(f'🎉 앗! 오늘이 바로 {ctx.author.name}님의 생일이네요! 정말정말 축하해요!! 🎂🎁🎊 생일빵 대신 제 하트를 받으세요! 💖 다들 생일축하해주세요🎊')
            
    else:
        await ctx.send(f'😥 아직 등록된 생일이 없어요. `!생일등록 MM-DD` (예: !생일등록 03-10) 명령어로 등록해 주세요!')

# --- 📜 생일 목록 (날짜 순서대로 정렬!) ---
@bot.command(name='생일목록')
async def list_birthdays(ctx):
    # 몽고DB에서 모든 유저를 가져오는데, 'date'를 기준으로 오름차순(1) 정렬!
    users = collection.find().sort("date", 1)
    
    if collection.count_documents({}) == 0:
        await ctx.send("텅~ 아직 아무도 생일을 등록하지 않았어요!")
        return
    
    msg = "🎂 **우리 서버 생일 목록 (날짜순)** 🎂\n"
    msg += "---" * 10 + "\n"
    
    for user in users:
        # 09-06 같은 날짜 형식을 더 보기 좋게 '9월 6일'로 바꾸고 싶다면 아래처럼 해도 돼!
        msg += f"• **{user['date']}** : {user['name']}님\n"
        
    await ctx.send(msg)

# --- ⏰ 매일 자정에 자동 생일 축하 ---
# 매일 밤 12시 정각(한국 시간)에 작동하도록 설정
@tasks.loop(time=time(hour=0, minute=0, tzinfo=KST))
async def check_birthday_loop():
    # 오늘 날짜 구하기 (예: "03-10")
    today_str = datetime.now(KST).strftime("%m-%d")
    
    # 몽고DB에서 오늘 날짜랑 생일이 똑같은 사람 찾기
    birthday_users = list(collection.find({"date": today_str}))
    
    # 생일인 사람이 있다면 축하 메시지 보내기!
    if len(birthday_users) > 0:
        for guild in bot.guilds:
            # 서버의 기본 안내 채널이나, 없으면 첫 번째 일반 채팅방을 찾음
            channel = guild.system_channel
            if not channel:
                channel = guild.text_channels[0]
            
            if channel:
                for user in birthday_users:
                    await channel.send(f"🎉 **축하합니다!** 오늘은 <@{user['_id']}>님의 생일이에요! 모두 모여서 축하해 주세요! 🎂🎁")

# --- 🔑 디스코드 봇 켜기 ---
TOKEN = os.environ.get('BOT_TOKEN')
bot.run(TOKEN)
