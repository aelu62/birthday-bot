import discord
from discord.ext import commands, tasks
import os
from pymongo import MongoClient
from datetime import datetime, time, timezone, timedelta

# --- ⚙️ 기본 및 시간 설정 ---
KST = timezone(timedelta(hours=9))
intents = discord.Intents.default()
intents.message_content = True
bot = commands.Bot(command_prefix='!', intents=intents)

# --- ⚙️ 몽고DB(비밀 금고) 연결 ---
MONGO_URL = os.environ.get('MONGO_URL')
client = MongoClient(MONGO_URL)
db = client['birthday_db']
collection = db['users']
settings_collection = db['settings'] # 💡 채널 설정을 저장할 새로운 금고 공간!

# --- 🚀 봇이 켜졌을 때 ---
@bot.event
async def on_ready():
    print(f'✅ 로그인 성공: {bot.user.name}')
    if not check_birthday_loop.is_running():
        check_birthday_loop.start()

# --- 🛠️ 생일 축하 채널 설정 (👑 관리자 전용) ---
@bot.command(name='생일채널설정')
@commands.has_permissions(administrator=True) # 관리자 권한이 있는 사람만 허용!
async def set_birthday_channel(ctx, channel: discord.TextChannel = None):
    # 채널을 멘션하지 않으면, 명령어를 친 현재 채널을 기본으로 설정해.
    if channel is None:
        channel = ctx.channel
        
    # 몽고DB에 서버 아이디와 채널 아이디를 단단히 저장!
    settings_collection.update_one(
        {'_id': str(ctx.guild.id)},
        {'$set': {'channel_id': str(channel.id)}},
        upsert=True
    )
    await ctx.send(f'📢 띠링! 앞으로 자동 생일 축하 메시지는 {channel.mention} 채널에만 올라옵니다! (관리자 설정 완료 🛠️)')

# 관리자가 아닌 사람이 명령어를 쳤을 때 차단하는 기능
@set_birthday_channel.error
async def set_birthday_channel_error(ctx, error):
    if isinstance(error, commands.MissingPermissions):
        await ctx.send("🚫 삐빅! 이 명령어는 서버 관리자만 사용할 수 있어요!")

# --- 🎂 생일 등록 (중복 방지 기능 추가!) ---
@bot.command(name='생일등록')
async def register_birthday(ctx, date: str):
    user_id = str(ctx.author.id)
    
    # 1. 이미 금고에 내 데이터가 있는지 먼저 슬쩍 확인!
    existing_user = collection.find_one({'_id': user_id})
    if existing_user:
        await ctx.send(f'⚠️ {ctx.author.name}님은 이미 생일(**{existing_user["date"]}**)이 등록되어 있어요! 변경을 원하시면 `!생일변경 MM-DD` 명령어를 사용해 주세요.')
        return # 등록 중단!
        
    # 2. 등록된 적이 없다면 몽고DB에 저장!
    collection.update_one(
        {'_id': user_id}, 
        {'$set': {'date': date, 'name': ctx.author.name}}, 
        upsert=True
    )
    
    # 3. 잘 들어갔는지 확인!
    check_data = collection.find_one({'_id': user_id})
    if check_data:
        await ctx.send(f'🎉 {check_data["name"]}님의 생일(**{check_data["date"]}**)변경 완료')
    else:
        await ctx.send('⚠️ 에러: 몽고DB 서버에 데이터를 찾지 못했습니다.')

# --- 🔄 생일 변경 (수정 전용 명령어!) ---
@bot.command(name='생일변경')
async def change_birthday(ctx, date: str):
    user_id = str(ctx.author.id)
    
    # 1. 진짜 등록된 적이 있는지 확인!
    existing_user = collection.find_one({'_id': user_id})
    if not existing_user:
        await ctx.send(f'😥 아직 등록된 생일이 없어요. 변경할 생일이 없으니 `!생일등록 MM-DD` 명령어로 먼저 등록해 주세요!')
        return
        
    # 2. 등록된 적이 있다면 새로운 날짜로 업데이트!
    collection.update_one(
        {'_id': user_id}, 
        {'$set': {'date': date, 'name': ctx.author.name}}
    )
    
    await ctx.send(f'🔄 {ctx.author.name}님의 생일이 **{date}**(으)로 성공적으로 변경되었습니다! 몽고DB 업데이트 완료! 🛡️')


# --- 🎂 생일 확인 (내 생일) ---
@bot.command(name='생일확인')
async def check_birthday(ctx):
    user_id = str(ctx.author.id)
    user_data = collection.find_one({'_id': user_id})
    
    if user_data:
        await ctx.send(f'📅 {ctx.author.name}님의 등록된 생일은 **{user_data["date"]}** 입니다!')
        today_str = datetime.now(KST).strftime("%m-%d")
        if user_data["date"] == today_str:
            await ctx.send(f'🎉 앗! 오늘이 바로 {ctx.author.name}님의 생일이네요! 정말정말 축하해요!! 🎂🎁🎊 생일빵 대신 제 하트를 받으세요! 💖 모두 축하해주세요')
    else:
        await ctx.send(f'😥 아직 등록된 생일이 없어요. `!생일등록 MM-DD` (예: !생일등록 03-10) 명령어로 등록해 주세요!')

# --- 📜 생일 목록 (깔끔한 버전) ---
@bot.command(name='생일목록')
async def list_birthdays(ctx):
    users = collection.find().sort("date", 1)
    if collection.count_documents({}) == 0:
        await ctx.send("🎂 아직 등록된 생일이 없어요!")
        return
    
    msg = "🎂 **우리 서버 생일 목록** 🎂\n"
    msg += "━━━━━━━━━━━━━━━━━━\n"
    for user in users:
        msg += f"• **{user['date']}** : {user['name']}님\n"
    await ctx.send(msg)

# --- ⏰ 매일 자정에 자동 생일 축하 (지정된 채널에만!) ---
@tasks.loop(time=time(hour=0, minute=0, tzinfo=KST))
async def check_birthday_loop():
    today_str = datetime.now(KST).strftime("%m-%d")
    birthday_users = list(collection.find({"date": today_str}))
    
    if len(birthday_users) > 0:
        for guild in bot.guilds:
            # 몽고DB에서 이 서버에 '관리자가 설정한 채널'이 있는지 확인!
            guild_setting = settings_collection.find_one({'_id': str(guild.id)})
            
            channel = None
            if guild_setting and 'channel_id' in guild_setting:
                channel = guild.get_channel(int(guild_setting['channel_id']))
            else:
                # 설정된 채널이 없으면 기본 시스템 채널로 보냄
                channel = guild.system_channel
                if not channel:
                    channel = guild.text_channels[0]
            
            if channel:
                for user in birthday_users:
                    await channel.send(f"🎉 **축하합니다!** 오늘은 <@{user['_id']}>님의 생일이에요! 모두 모여서 축하해 주세요! 🎂🎁")

# --- 🔑 디스코드 봇 켜기 ---
TOKEN = os.environ.get('BOT_TOKEN')
bot.run(TOKEN)
